Convertstring

Mar 20th, 2014 

    char * p = "abcdefghi";
        
    unsigned long len = (unsigned long)strlen((const char*)p);
    unsigned long myl = len / 2;
                    
    char * k = (char *)malloc(len);
    k = strdup((const char*)p);

    for (int i = 0; i < myl; i++) {
        char c = p[i];
        memcpy(k+i,(p+len-i-1),1);
        memcpy((k+len-i-1),&c,1);
    }
                                                                        
    p = k;
    free(k);
    k = NULL;
    printf("%s\n",p);
